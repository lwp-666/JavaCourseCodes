#####  用户信息表

```
CREATE TABLE `customer`(
	customer_id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY NOT NULL COMMENT '主键id',
	customer_name VARCHAR(20) NOT NULL COMMENT '用户姓名',
	identity_type TINYINT NOT NULL DEFAULT 1 COMMENT '证据类型：1.身份证，2.军官证，3.护照',
	identity_no VARCHAR(20) COMMENT '证件号码',
	mobile_phone INT UNSIGNED COMMENT '手机号',
	email VARCHAR(30) COMMENT '邮箱',
	gender CHAR(1) COMMENT '性别',
	register_time TIMESTAMP NOT NULL COMMENT '注册时间',
	birthday DATETIME COMMENT '会员生日',
	user_money DECIMAL(8,2) NOT NULL DEFAULT 0.00 COMMENT '账户余额'
)ENGINE = innodb COMMENT '用户信息表';
```

##### 订单表.
```
CREATE TABLE `t_order`  (
  `id` int UNSIGNED PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
  `code` varchar(200) NOT NULL COMMENT '流水号',
  `type` tinyint UNSIGNED NOT NULL COMMENT '订单类型：1实体销售,2网络销售',
  `shop_id` int UNSIGNED COMMENT '零售店ID',
  `customer_id` int UNSIGNED COMMENT '会员ID',
  `amount` decimal(10, 2) UNSIGNED NOT NULL COMMENT '总金额',
  `payment_type` tinyint UNSIGNED NOT NULL COMMENT '支付方式：1借记卡,2信用卡,3微信,4支付宝,5现金',
  `status` tinyint UNSIGNED NOT NULL COMMENT '状态：1未付款,2已付款,3已发货,4已签收',
  `postage` decimal(10, 2) UNSIGNED COMMENT '邮费',
  `weight` int UNSIGNED COMMENT '重量（克）',
  `voucher_id` int UNSIGNED COMMENT '购物券ID',
  `create_time` timestamp NOT NULL DEFAULT NOW() COMMENT '创建时间'
) COMMENT = '订单表';
```

##### 商品信息表
```
CREATE TABLE `product_info`(
  `product_id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY NOT NULL COMMENT '商品ID',
  `product_core` CHAR(16) NOT NULL COMMENT '商品编码',
  `product_name` VARCHAR(20) NOT NULL COMMENT '商品名称',
  `brand_id` INT UNSIGNED NOT NULL COMMENT '品牌表的ID',
  `supplier_id` INT UNSIGNED NOT NULL COMMENT '商品的供应商ID',
  `price` DECIMAL(8,2) NOT NULL COMMENT '商品销售价格',
  `publish_status` TINYINT NOT NULL DEFAULT 0 COMMENT '上下架状态：0下架1上架',
  `production_date` DATETIME NOT NULL COMMENT '生产日期',
  `shelf_life` INT NOT NULL COMMENT '商品有效期',
  `descript` TEXT NOT NULL COMMENT '商品描述',
  `indate` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '商品录入时间'
) ENGINE = innodb COMMENT '商品信息表';
```
